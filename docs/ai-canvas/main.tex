        \textbf{Box 2: Context \& Project Type} \newline 
        \textcolor{red}{\scriptsize Classify as Research (truth-seeking), Tech Development (usable prototype), or Product work (user value).} \\ \hline

        \textcolor{red}{\scriptsize Specify source (internal/external), formats (structured/unstructured), and quality dimensions (completeness, accuracy, timeliness).} &

        \textcolor{red}{\scriptsize Select method family: Supervised (Classification/Regression), Unsupervised (Clustering), RL, or GenAI. Justify why it fits the data[cite: 507].} \\ \hline
        
        \textbf{Box 5: ML Pipeline (DSPRO2)} \newline 
        \textcolor{red}{\scriptsize Map the end-to-end flow: Collection $\rightarrow$ Preparation $\rightarrow$ Model Training $\rightarrow$ Evaluation $\rightarrow$ Deployment.} & 
        \textbf{Box 6: Success Definition} \newline 
        \textcolor{red}{\scriptsize Pick one technical task metric (e.g., F1-score, MAE) and one stakeholder KPI (e.g., time saved, revenue).} \\ \hline
        
        \textbf{Box 7: Risks \& Guardrails} \newline 
        \textcolor{red}{\scriptsize Assess risk level (EU AI Act). Address potential bias or privacy issues and define human-in-the-loop oversight.} & 
        \textbf{Box 8: Infrastructure (DSPRO2)} \newline 
        \textcolor{red}{\scriptsize Strategy: Cloud (scalability/speed) vs. On-Premises (control/governance). Justify based on workload.} \\ \hline
        
        \textbf{Box 9: Costs \& Timeline (DSPRO2)} \newline 
        \textcolor{red}{\scriptsize Estimate GPU compute, storage, and data labeling costs[cite: 2506]. Provide a rough timeline (e.g., 6--12 months).} & 
        \textbf{Box 10: Team \& Next Steps (DSPRO2)} \newline 
        \textcolor{red}{\scriptsize Assign roles (Project Owner, Domain Expert, ML Engineer)[cite: 2395]. Define the first concrete action to begin execution.} \\ \hline
